Animation Generator

Create .MOV files with alpha channel for video editing

Python Generator Code

import numpy as np
import matplotlib.pyplot as plt
import matplotlib.animation as animation
from matplotlib.colors import to_rgba
import imageio

# Set up the figure with transparent background
fig = plt.figure(figsize=(8, 8), facecolor='none')
ax = fig.add_axes([0, 0, 1, 1], frameon=False)
ax.set_xlim(0, 1), ax.set_xticks([])
ax.set_ylim(0, 1), ax.set_yticks([])

# Create objects that will change over time
circles = []
colors = ['red', 'green', 'blue', 'yellow', 'purple']
for i in range(5):
    circle = plt.Circle((0.2 + i*0.15, 0.5), 0.1, 
                       fc=to_rgba(colors[i], alpha=0.7), 
                       ec='white', lw=2)
    ax.add_patch(circle)
    circles.append(circle)

text = ax.text(0.5, 0.15, "ANIMATION", 
              ha='center', va='center', 
              fontsize=48, fontweight='bold',
              color='white')

def update(frame):
    for i, circle in enumerate(circles):
        y_offset = 0.1 * np.sin(frame * 0.1 + i)
        circle.set_center((circle.center[0], 0.5 + y_offset))
        
        current_color = circle.get_facecolor()
        new_alpha = (np.sin(frame * 0.05 + i) + 1) / 2 * 0.7
        circle.set_facecolor(to_rgba(colors[i], alpha=new_alpha))
    
    scale = 1 + 0.2 * np.sin(frame * 0.1)
    text.set_transform(ax.transData + 
                      plt.Affine2D().scale(scale, scale))
    
    hue = frame % 360
    text.set_color(f'hsv({hue}, 80%, 100%)')
    return circles + [text]

ani = animation.FuncAnimation(fig, update, frames=100, blit=True)

print("Rendering animation...")
writer = imageio.get_writer('animation_with_alpha.mov', 
                          fps=30, 
                          codec='prores_ks',
                          pixelformat='yuva444p10le',
                          quality=10)

for frame in range(100):
    update(frame)
    fig.canvas.draw()
    image = np.frombuffer(fig.canvas.tostring_argb(), dtype=np.uint8)
    image = image.reshape(fig.canvas.get_width_height()[::-1] + (4,))
    image = np.roll(image, 3, axis=2)
    writer.append_data(image)
    print(f"Frame {frame + 1}/100 rendered", end='\r')

writer.close()
print("\nAnimation saved as 'animation_with_alpha.mov'")
