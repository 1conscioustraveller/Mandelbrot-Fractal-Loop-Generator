packages = ["matplotlib"] from js import document import io, base64, math import matplotlib.pyplot as plt # Persistent figure & axes fig, ax = plt.subplots(figsize=(6, 4)) fig.patch.set_facecolor("black") ax.set_facecolor("black") # Globals mode = "Solve I" V = 9.0 I = 0.09 R = 100.0 def get_value(id): return float(document.getElementById(id).value) def get_select_value(id): return document.getElementById(id).value def set_image(id, data_url): document.getElementById(id).src = data_url def set_text(id, text): document.getElementById(id).innerText = text def draw_battery(ax, x, y, height, spacing): ax.plot([x - spacing/2, x - spacing/2], [y - height, y + height], color="white") ax.plot([x + spacing/2, x + spacing/2], [y - height/2, y + height/2], color="white") def draw_resistor(ax, x1, y1, x2, y2, zigzags=6, amplitude=0.25): dx = x2 - x1 dy = y2 - y1 length = (dx**2 + dy**2)**0.5 if length == 0: return ux, uy = dx/length, dy/length px, py = -uy, ux pts = [(x1, y1)] segment = length / (zigzags * 2) for i in range(1, zigzags * 2): t = i * segment bx = x1 + ux * t by = y1 + uy * t offset = amplitude if i % 2 == 1 else -amplitude pts.append((bx + px * offset, by + py * offset)) pts.append((x2, y2)) xs, ys = zip(*pts) ax.plot(xs, ys, color="white") def format_val(val, unit): if val is None or (isinstance(val, float) and (math.isnan(val) or math.isinf(val))): return "—" if abs(val) >= 100: s = f"{val:.0f}" elif abs(val) >= 10: s = f"{val:.1f}" else: s = f"{val:.2f}" return f"{s} {unit}" def render(): left, right, top, bottom = -4, 4, 1.5, -1.5 ax.cla() ax.set_facecolor("black") # Wires ax.plot([left, right], [top, top], color="white") ax.plot([right, right], [top, bottom], color="white") ax.plot([left, right], [bottom, bottom], color="white") ax.plot([left, left], [bottom, top], color="white") # Elements draw_battery(ax, left, 0.0, height=0.8, spacing=0.6) draw_resistor(ax, 0.0, 0.0, right, 0.0) lw = 1.0 + 4.0 * (0.0 if math.isnan(I) else min(1.0, abs(I)/2.0)) ax.annotate("", xy=(right - 0.5, top), xytext=(left + 0.5, top), arrowprops=dict(arrowstyle="->", linewidth=lw, color="cyan")) ax.text(0.0, top + 0.3, f"I = {format_val(I, 'A')}", ha="center", va="bottom", color="cyan") ax.text(left - 0.2, 0.0, f"V = {format_val(V, 'V')}", ha="right", va="center", rotation=90, color="green") ax.text(right + 0.2, 0.0, f"R = {format_val(R, 'Ω')}", ha="left", va="center", color="yellow") eq = r"$I = \dfrac{V}{R}$" if mode == "Solve I" else (r"$V = I \cdot R$" if mode == "Solve V" else r"$R = \dfrac{V}{I}$") ax.text(0.0, bottom - 0.4, eq, ha="center", va="top", color="white") ax.set_aspect("equal") ax.set_xlim(left - 1.5, right + 1.5) ax.set_ylim(bottom - 1.0, top + 1.0) ax.axis("off") buf = io.BytesIO() fig.savefig(buf, format="png", bbox_inches="tight", facecolor="black") buf.seek(0) img_base64 = base64.b64encode(buf.read()).decode("utf-8") set_image("circuit-img", f"data:image/png;base64,{img_base64}") # Power display if not (math.isnan(V) or math.isnan(I)): power = V * I set_text("power-display", f"Power: {format_val(power, 'W')}") else: set_text("power-display", "Power: —") def update_plot(event=None): global mode, V, I, R mode = get_select_value("mode-select") V = get_value("slider-V") I = get_value("slider-I") R = get_value("slider-R") if mode == "Solve I": I = float('nan') if R <= 0 else V / R elif mode == "Solve V": V = I * R elif mode == "Solve R": R = float('nan') if I <= 0 else V / I render() # Bind events document.getElementById("slider-V").oninput = update_plot document.getElementById("slider-I").oninput = update_plot document.getElementById("slider-R").oninput = update_plot document.getElementById("mode-select").onchange = update_plot # Initial render render()
